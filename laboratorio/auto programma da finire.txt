import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        
        // Creo un auto
        System.out.println("=== CREA LA TUA AUTO ===\n");
        
        System.out.print("Marca dell' auto: ");
        String marca = input.nextLine();
        
        System.out.print("Modello: ");
        String modello = input.nextLine();
        
        // Creo l'oggetto usando il costruttore
        Auto mioAuto = new Auto(marca, modello);
        
        // Uso i metodi
        System.out.println("\n--- AZIONI DELL' AUTO ---\n");
        mioAuto.mostraInfo();
        mioAuto.accendi();
        mioAuto.chiama("Mario");
        mioAuto.mandaMessaggio("Ciao!");
        mioAuto.spegni();
        
        input.close();
    }
}
